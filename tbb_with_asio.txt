tbb::empty_task* StartServer(tbb_asio::asio_server * pServer) {
	tbb::empty_task* parent = new (tbb::task::allocate_root()) tbb::empty_task;

	RunIOServer * s = new (parent->allocate_child()) RunIOServer(pServer);
	parent->spawn(*s);
	return parent;

#include "stdafx.h"
#include "asio_server.h"
#include <boost/bind.hpp>
#include <boost/shared_ptr.hpp>

namespace tbb_asio
{
	asio_server::asio_server( const std::string& address, const
std::string& port )
		: acceptor_(io_service_), new_connection_(new
connection(io_service_, request_handler_))
	{
		// Open the acceptor with the option to reuse the address
		asio::ip::tcp::resolver resolver(io_service_);
		asio::ip::tcp::resolver::query query(address,port);
		asio::ip::tcp::endpoint endpoint = *resolver.resolve(query);
		//asio::ip::tcp::endpoint endpoint( asio::ip::tcp::v4(), 4444 );
		//endpoint./*.address(ADDR_ANY)*/;
		acceptor_.open(endpoint.protocol());
		acceptor_.set_option(asio::ip::tcp::acceptor::reuse_address(true));
		acceptor_.bind(endpoint);
		acceptor_.listen();
		acceptor_.async_accept(new_connection_->socket(),
boost::bind(&asio_server::handle_accept, this,
asio::placeholders::error));

	}

	void asio_server::stop()
	{
		io_service_.stop();

	}

	void asio_server::run()
	{
		io_service_.run();
	}

	void asio_server::handle_accept( const asio::error_code& e )
	{
		if(!e)
		{
			printf("Accepting new socket connection.\r\n");
			new_connection_->start();
			new_connection_.reset(new connection(io_service_, request_handler_));
			acceptor_.async_accept(new_connection_->socket(),
				boost::bind(&asio_server::handle_accept, this,
				asio::placeholders::error ));
		}
	}
}

void * operator new (size_t size) throw (std::bad_alloc)
{
	if( size == 0 ) size = 1;
	if( void * ptr = scalable_malloc(size))
		return ptr;
	throw std::bad_alloc();
}
void * operator new[] (size_t size) throw (std::bad_alloc)
{
	return operator new(size);
}
void * operator new(size_t size, const std::nothrow_t& ) throw()
{
	if( size == 0 ) size = 1;
	if( void * ptr = scalable_malloc(size))
		return ptr;
	return NULL;
}

void * operator new[] (size_t size, const std::nothrow_t&) throw()
{
	return operator new(size, std::nothrow );
}

void operator delete( void * ptr ) throw()
{
	if( ptr != 0 ) scalable_free(ptr);
}

void operator delete[] (void * ptr) throw()
{
	operator delete(ptr);
}
